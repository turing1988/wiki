 35 func! ShortTabLabel()
 36     let bufnrlist = tabpagebuflist(v:lnum)
 37     let label = bufname(bufnrlist[tabpagewinnr(v:lnum) -1])
 38     let filename = fnamemodify(label, ':t')
 39     return filename
 40 endfunc

141 
142     function! ClosePair(char)
143         if getline('.')[col('.') - 1] == a:char
144             return "\<Right>"
145         else
146             return a:char
147         endif
148     endf
149 
150     function CloseBracket()
151         if match(getline(line('.') + 1), '\s*}') < 0
152             return "\<CR>}"
153         else
154             return "\<Esc>j0f}a"
155         endif
156     endf
157 
158     function QuoteDelim(char)
159         let line = getline('.')
160         let col = col('.')
161         if line[col - 2] == "\\"
162             "Inserting a quoted quotation mark into the string
163             return a:char
164         elseif line[col - 1] == a:char
165             "Escaping out of the string
166             return "\<Right>"
167         else
168             "Starting a string
169             return a:char.a:char."\<Esc>i"
170         endif
171     endf
172 endif

206 map <F2> :call Do_CsTag() <cr>:redraw!<cr>
207 function! Do_CsTag()
208     let dir = getcwd()
209     if has("cscope")
210         silent! execute "cs kill -1"
211     endif
212     if(executable('ctags'))
213         silent! execute "!ctags -R --c-types=+p --fields=+lS *"
214         " silent! execute "!ctags -R --c++-kinds=+p --fields=+iaS --extra=+q ."
215     endif
216     if(executable('cscope') && has("cscope") )
217         silent! execute "!find . -name '*.h' -o -name '*.c' -o -name '*.cpp' -o -name '*.java' -o -name '*.cs' > cscope.files"
218         silent! execute "!cscope -Rbq"
219         execute "normal :"
220     endif
221     set tags=./tags      "将当前的工程的tags导入
222     set tags+=~/.vim/systags "将系统已经生成的tags导入
223     set csprg=/usr/bin/cscope
224     set cst    "同时查找tags cscope
225     set csto=1 "优先查找tags
226     set nocsverb
227     if filereadable("cscope.out")
228         exe "cs add cscope.out"
229     elseif $CSCOPE_DB != ""
230         exe "cs add $CSCOPE_DB"
231     endif
232     set csverb
233     set cst
234 endf

307 "定义FormartSrc()
308 func! FormartSrc()
309     exec "w"
310     "C程序,Perl程序,Python程序
311     if &filetype == 'c'
312         exec "!astyle --style=ansi -U -p --break-elseifs  --suffix=none %"
313         exec "e! %"
314     elseif &filetype == 'cpp'
315         exec "!astyle -q --style=ansi -U  -P --break-elseifs  --suffix=none %"
316         "exec "e! %"
317     elseif &filetype == 'perl'
318         exec "!astyle --style=ansi --suffix=none %"
319         exec "e! %"
320     elseif &filetype == 'py'
321         exec "!astyle --style=ansi --suffix=none %"
322         exec "e! %"
323         "Java程序
324     elseif &filetype == 'java'
325         exec "!astyle --style=java --suffix=none %"
326         exec "e! %"
327     elseif &filetype == 'jsp'
328         exec "!astyle --style=ansi --suffix=none %"
329         exec "e! %"
330     elseif &filetype == 'xml'
331         exec "!astyle --style=ansi --suffix=none %"
332         exec "e! %"
333     elseif &filetype == 'html'
334         exec "!astyle --style=ansi --suffix=none %"
335         exec "e! %"
336     elseif &filetype == 'htm'
337         exec "!astyle --style=ansi --suffix=none %"
338         exec "e! %"
339     endif
340     return
